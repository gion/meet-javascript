a good  jQuery plugin must

  • maintain chainability
  • have consistent results
  • be written using "jquery standards"
    (I didn't know what else to call'em)


what... huh? :-??

maintain chainability

= return the context that your plugin has received 
$.fn.somePlugin = function(){
    // bla bla bla logic

    // this makes the magic happen
    return this;
}
this means that the chain won't break: 
$('div').somePlugin().text("we're chained, bro!");

consistent results

= apply the same logic on any element 
$.fn.countElements = function(){
    var counter = 0,
        countElement = function(el) {
            counter++;
            $(el).text("I'm element #" + counter);
        };

    // separate logic applied for EVERY element
    this.each(function(i, el){
        countElement(el);
    });

    return this;
}

here's a demo : http://codepen.io/gion/pen/wdxgy

"jquery standards"

= at least pass $ as reference to jQuery through a closure 
var $ = 'something else than jQuery';
(function($){
    // I can use $ as jQuery here :)
    // take that, wordpress!

    $.fn.somePlugin = function(){ /*...*/ }
})(jQuery);

the good version of our first plugin

//closure 
(function($){
    $.fn.vCenter = function(){

        // each is chainable, so we can return it directly ;) 
        return this.each(function(i, element){
            // anyone has the right to find it's own center

            var el = $(this),
                parent = el.parent(),
                parentHeight = parent.height(),
                height = el.height(),
                desiredOffset = (parentHeight - height) / 2;

            el.css('margin-top', desiredOffset);
        });
    }
})(jQuery);
